 public List<IconType<FilterType<T>>> getIconList()
   {
      List<IconType<FilterType<T>>> list = new ArrayList<IconType<FilterType<T>>>();      
      List<Node> iconListNodes = childNode.get(IconTypeImpl.nodeName);
      for(Node node: iconListNodes)
      {
         IconTypeImpl<FilterType<T>> iconType = new IconTypeImpl<FilterType<T>>(this, "", childNode, node);
         list.add(iconType);

      return list;
